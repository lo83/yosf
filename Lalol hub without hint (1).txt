    local idOnly = tostring(soundId):match("%d+")
    local entry = Instance.new("Frame", tScroll); entry.Size = UDim2.new(1, 0, 0, 36); entry.BackgroundColor3 = Color3.fromRGB(25, 25, 30); Instance.new("UICorner", entry)
    local info = Instance.new("TextLabel", entry); info.Size = UDim2.new(0.55, 0, 1, 0); info.Position = UDim2.new(0, 6, 0, 0); info.Text = "Owner: "..ownerName.."\nID: "..idOnly; info.TextColor3 = Color3.new(1,1,1); info.Font = Enum.Font.Gotham; info.TextSize = 9; info.BackgroundTransparency = 1; info.TextXAlignment = Enum.TextXAlignment.Left
    local cp = createBtn(entry, "Copy", Color3.fromRGB(50, 50, 50), UDim2.new(0.6, 0, 0.2, 0), UDim2.new(0, 45, 0, 20))
    cp.MouseButton1Click:Connect(function() if setclipboard then setclipboard(idOnly) cp.Text = "Done" task.wait(0.4) cp.Text = "Copy" end end)
    local ls = createBtn(entry, "Listen", Color3.fromRGB(0, 100, 0), UDim2.new(0.8, 0, 0.2, 0), UDim2.new(0, 45, 0, 20))
    ls.MouseButton1Click:Connect(function()
        if currentPreview then currentPreview:Stop(); currentPreview:Destroy() end
        if ls.Text == "Listen" then for _, v in pairs(tScroll:GetDescendants()) do if v:IsA("TextButton") and v.Text == "Stop" then v.Text = "Listen"; v.BackgroundColor3 = Color3.fromRGB(0, 100, 0) end end
            ls.Text = "Stop"; ls.BackgroundColor3 = Color3.fromRGB(130, 0, 0); currentPreview = Instance.new("Sound", SoundService); currentPreview.SoundId = "rbxassetid://" .. idOnly; currentPreview:Play()
        else ls.Text = "Listen"; ls.BackgroundColor3 = Color3.fromRGB(0, 100, 0) end
    end)
end

local function checkSound(v) if v:IsA("Sound") and v.SoundId ~= "" then local id = v.SoundId:match("%d+") if id and not lastSeen[id] then lastSeen[id] = true createTrackerEntry(id, "Player") end end end
workspace.DescendantAdded:Connect(function(v) task.wait(0.1) checkSound(v) end)

    if antiVoidActive then pcall(function() local char = plr.Character; local hrp = char and char:FindFirstChild("HumanoidRootPart") if hrp and char.Humanoid.Health > 0 and hrp.Position.Y < -50 then hrp.Velocity = Vector3.new(0, 0, 0); hrp.CFrame = CFrame.new(Vector3.new(0, 20, 0)) end end) end
    if IsActive and TargetPlayer and TargetPlayer.Character and plr.Character then
        local MyRoot = plr.Character:FindFirstChild("HumanoidRootPart"); local TheirRoot = TargetPlayer.Character:FindFirstChild("HumanoidRootPart")
        if MyRoot and TheirRoot then local Speed = tonumber(SpeedInput.Text) or 25; Angle = Angle + (Speed * dt); local MoveOffset = math.sin(Angle) * 1.5
            if Mode == "Front" then MyRoot.CFrame = TheirRoot.CFrame * CFrame.new(0, 0.9, -1.1) * CFrame.Angles(0, math.pi, 0) * CFrame.new(0, 0, MoveOffset)
            elseif Mode == "Back" then MyRoot.CFrame = TheirRoot.CFrame * CFrame.new(0, 0.9, 1.1) * CFrame.new(0, 0, MoveOffset) end MyRoot.Velocity = Vector3.new(0,0,0)
        end
    end
end)
