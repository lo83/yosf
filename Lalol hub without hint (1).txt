    info.Text = "Owner: " .. ownerName .. "\nID: " .. idOnly
    info.TextColor3 = Color3.new(1, 1, 1)
    info.TextXAlignment = Enum.TextXAlignment.Left

    cpBtn.Size = UDim2.new(0, 60, 0, 22)

    cpBtn.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    cpBtn.TextColor3 = Color3.new(1, 1, 1)
    Instance.new("UICorner", cpBtn)

    cpBtn.MouseButton1Click:Connect(function()
        if setclipboard then setclipboard(idOnly) cpBtn.Text = "Done" task.wait(0.5) cpBtn.Text = "Copy" end
    end)

    local lsBtn = Instance.new("TextButton", entry)
    lsBtn.Size = UDim2.new(0, 60, 0, 22)
    lsBtn.Position = UDim2.new(0.75, -5, 0.55, 0)
    lsBtn.Text = "Listen"
    lsBtn.BackgroundColor3 = Color3.fromRGB(0, 100, 0)
    lsBtn.TextColor3 = Color3.new(1, 1, 1)
    Instance.new("UICorner", lsBtn)

    lsBtn.MouseButton1Click:Connect(function()
        if currentPreview then currentPreview:Stop() currentPreview:Destroy() end
        if lsBtn.Text == "Listen" then
            for _, v in pairs(scroll:GetDescendants()) do
                if v:IsA("TextButton") and v.Text == "Stop" then v.Text = "Listen" v.BackgroundColor3 = Color3.fromRGB(0, 100, 0) end
            end

local function checkSound(v)
    if v:IsA("Sound") and v.SoundId ~= "" then
        local id = v.SoundId:match("%d+")
        if id and not lastSeen[id] then lastSeen[id] = true createTrackerEntry(id, "Player") end
    end
end

workspace.DescendantAdded:Connect(function(v) task.wait(0.1) checkSound(v) end)
for _, v in pairs(workspace:GetDescendants()) do checkSound(v) end

-- [[ Logic & Teleport Functions ]]
local TargetPlayer = nil
local Mode = ""
local IsActive = false
local Angle = 0
local antiVoidActive = true
local invisOn, ghostOn, noclipOn, qonataActive = false, false, false, false
local ghostSpeed = 50

local function GetPlayer(S)
    if S == "" then return nil end
    for _, v in pairs(Players:GetPlayers()) do
        if v.Name:lower():find(S:lower()) or v.DisplayName:lower():find(S:lower()) then return v end
    end
end

-- لۆجیکێ گۆڕینا خێرایی و بازدانێ
local function updateStats()
    local char = plr.Character
    local hum = char and char:FindFirstChildOfClass("Humanoid")
    if hum then
        local speedVal = tonumber(CustomSpeedInput.Text)
        local jumpVal = tonumber(CustomJumpInput.Text)
        if speedVal then hum.WalkSpeed = speedVal end
        if jumpVal then hum.JumpPower = jumpVal; hum.UseJumpPower = true end
    end
end

ApplyStatsBtn.MouseButton1Click:Connect(updateStats)

    char:WaitForChild("Humanoid")
    task.wait(0.5)
    updateStats()
end)

local function setTransparency(char, val)
    for _, p in ipairs(char:GetDescendants()) do
        if p:IsA("BasePart") and p.Name ~= "HumanoidRootPart" then p.Transparency = val end
    end
end

-- Loop Heartbeat updates
RunService.Heartbeat:Connect(function(dt)
    -- Anti-Void
    if antiVoidActive then
        pcall(function()
            local char = plr.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            if hrp and char.Humanoid.Health > 0 and hrp.Position.Y < -50 then
                hrp.Velocity = Vector3.new(0, 0, 0)
                hrp.CFrame = CFrame.new(Vector3.new(0, 20, 0))
            end
        end)
    end
    
    -- Xas Mod Motion
    if IsActive and TargetPlayer and TargetPlayer.Character and plr.Character then
        local MyRoot = plr.Character:FindFirstChild("HumanoidRootPart")
        local TheirRoot = TargetPlayer.Character:FindFirstChild("HumanoidRootPart")
        if MyRoot and TheirRoot then
            local Speed = tonumber(SpeedInput.Text) or 25
            Angle = Angle + (Speed * dt)
            local MoveOffset = math.sin(Angle) * 1.5
            
            if Mode == "Front" then
                MyRoot.CFrame = TheirRoot.CFrame * CFrame.new(0, 0.9, -1.1) * CFrame.Angles(0, math.pi, 0) * CFrame.new(0, 0, MoveOffset)
            elseif Mode == "Back" then
                MyRoot.CFrame = TheirRoot.CFrame * CFrame.new(0, 0.9, 1.1) * CFrame.new(0, 0, MoveOffset)
            end
            MyRoot.Velocity = Vector3.new(0,0,0)
        end
    end
end)
